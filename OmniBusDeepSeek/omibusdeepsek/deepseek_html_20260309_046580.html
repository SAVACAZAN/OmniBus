Script 1: Extracție Grid-DSL
Descarcă automat codul din toate repo-urile
GitHub API JSONL Dataset Auto-extract 
import os
import requests
import json
from pathlib import Path

# CONFIGURARE
ORG_NAME = "Grid-DSL"
GITHUB_TOKEN = "YOUR_GITHUB_TOKEN_HERE"  # Generează token la https://github.com
EXTENSII = {'.grid', '.py', '.js', '.ts', '.md', '.json', '.stat', '.rs', '.go', '.java', '.cpp', '.h'}
OUTPUT_FILE = "grid_dsl_dataset.jsonl"

headers = {"Authorization": f"token {GITHUB_TOKEN}"} if GITHUB_TOKEN else {}

def get_all_repos():
    """Obține toate repo-urile din organizație"""
    repos = []
    page = 1
    while True:
        url = f"https://api.github.com/orgs/{ORG_NAME}/repos?per_page=100&page={page}"
        response = requests.get(url, headers=headers)
        if response.status_code != 200:
            break
        data = response.json()
        if not data:
            break
        repos.extend(data)
        page += 1
    return repos

def get_repo_contents(repo_name, path=""):
    """Recursive fetch pentru toate fișierele"""
    contents_url = f"https://api.github.com/repos/{ORG_NAME}/{repo_name}/contents/{path}"
    response = requests.get(contents_url, headers=headers)
    if response.status_code != 200:
        return []
    
    files = []
    for item in response.json():
        if item['type'] == 'dir':
            files.extend(get_repo_contents(repo_name, item['path']))
        elif any(item['name'].endswith(ext) for ext in EXTENSII):
            files.append({
                'name': item['name'],
                'path': item['path'],
                'download_url': item['download_url'],
                'repo': repo_name
            })
    return files

def create_dataset():
    """Generează dataset-ul pentru fine-tuning"""
    print(f"🔍 Scanăm organizația {ORG_NAME}...")
    repos = get_all_repos()
    print(f"✅ Găsit {len(repos)} repo-uri")
    
    with open(OUTPUT_FILE, 'w', encoding='utf-8') as f:
        total_files = 0
        for repo in repos:
            repo_name = repo['name']
            print(f"📁 Procesez: {repo_name}")
            
            files = get_repo_contents(repo_name)
            for file_info in files:
                try:
                    content = requests.get(file_info['download_url']).text
                    
                    entry = {
                        "instruction": f"Implementează sau analizează codul Grid DSL din fișierul {file_info['name']}",
                        "input": f"Repo: {repo_name}\nContext: {file_info['path']}",
                        "output": content
                    }
                    f.write(json.dumps(entry, ensure_ascii=False) + '\n')
                    total_files += 1
                    
                    if total_files % 10 == 0:
                        print(f"   ⏱️  Procesat {total_files} fișiere...")
                        
                except Exception as e:
                    print(f"   ⚠️  Eroare la {file_info['name']}: {e}")
    
    print(f"\n✅ Dataset salvat: {OUTPUT_FILE}")
    print(f"📊 Total fișiere: {total_files}")
    
    # Statistici rapide
    with open(OUTPUT_FILE, 'r') as f:
        lines = f.readlines()
        total_size = sum(len(line) for line in lines) / 1024 / 1024
        print(f"📦 Dimensiune: {total_size:.2f} MB")
        print(f"📝 Exemple: {len(lines)}")

if __name__ == "__main__":
    if GITHUB_TOKEN == "YOUR_GITHUB_TOKEN_HERE":
        print("⚠️  ATENȚIE: Setează GITHUB_TOKEN în script!")
    else:
        create_dataset()

from unsloth import FastLanguageModel
import torch
from datasets import load_dataset
from trl import SFTTrainer
from transformers import TrainingArguments

# Configurare model
model_name = "unsloth/Qwen2.5-Coder-1B-Instruct-bnb-4bit"
max_seq_length = 2048

# Încărcare model cu optimizări pentru RTX 3070
model, tokenizer = FastLanguageModel.from_pretrained(
    model_name=model_name,
    max_seq_length=max_seq_length,
    load_in_4bit=True,
    dtype=None,  # Auto-detect
    device_map="auto",
)

# Configurare LoRA - specializare Grid DSL
model = FastLanguageModel.get_peft_model(
    model,
    r=64,  # Rank pentru capturarea specificului Grid DSL
    target_modules=["q_proj", "k_proj", "v_proj", "o_proj", "gate_proj", "up_proj", "down_proj"],
    lora_alpha=128,
    lora_dropout=0,
    bias="none",
    use_gradient_checkpointing="unsloth",  # CRITIC pentru 8GB VRAM
    random_state=42,
)

# Încărcare dataset (din Script 1)
dataset = load_dataset("json", data_files="grid_dsl_dataset.jsonl", split="train")

# Formatare prompt pentru Qwen
def format_chat(example):
    return {
        "text": tokenizer.apply_chat_template([
            {"role": "system", "content": "Ești un expert în Grid DSL, Crypto Trading și Statistică."},
            {"role": "user", "content": example['instruction'] + "\n" + example['input']},
            {"role": "assistant", "content": example['output']},
        ], tokenize=False)
    }

dataset = dataset.map(format_chat)

# Configurare antrenare
trainer = SFTTrainer(
    model=model,
    tokenizer=tokenizer,
    train_dataset=dataset,
    dataset_text_field="text",
    max_seq_length=max_seq_length,
    args=TrainingArguments(
        per_device_train_batch_size=2,  # Batch mic pentru 8GB
        gradient_accumulation_steps=4,
        warmup_steps=5,
        max_steps=100,  # Ajustează după mărimea datasetului
        learning_rate=2e-4,
        fp16=True,  # Folosește FP16 pe RTX 3070
        logging_steps=5,
        output_dir="grid_dsl_outputs",
        optim="adamw_8bit",  # Optimizare memorie
        save_strategy="steps",
        save_steps=50,
    ),
)

print("🚀 Începe antrenarea...")
trainer.train()

# Salvare model
model.save_pretrained("grid_dsl_model_final")
tokenizer.save_pretrained("grid_dsl_model_final")
print("✅ Antrenare completă! Model salvat în 'grid_dsl_model_final'")

Script 3: Test Matematic Crypto
Validare Z-Score cu NumPy (64-bit vs 1.5-bit)
Z-Score Cross-Validation Eroare Cuantizare 
import numpy as np
import torch
from unsloth import FastLanguageModel
import json

def calculate_zscore_numpy(prices, current_price):
    """Calcul Z-Score cu precizie dublă (64-bit)"""
    prices_array = np.array(prices, dtype=np.float64)
    mean = np.mean(prices_array)
    std = np.std(prices_array)
    z_score = (current_price - mean) / std
    return z_score, mean, std

def test_model_accuracy(model_path="grid_dsl_model_final"):
    """Testează acuratețea modelului pe calcule crypto"""
    
    # Încărcare model
    print("📦 Încărcare model...")
    model, tokenizer = FastLanguageModel.from_pretrained(
        model_path,
        load_in_4bit=True,
        device_map="auto"
    )
    FastLanguageModel.for_inference(model)
    
    # Date de test - volatilitate crypto
    test_cases = [
        {
            "prices": [62100, 62350, 61900, 62800, 63000, 62150, 61800, 62400, 62600, 62200],
            "current": 63500,
            "desc": "Breakout sus"
        },
        {
            "prices": [45000, 44800, 45200, 44900, 45100, 44700, 45300, 44800, 45000, 44900],
            "current": 44000,
            "desc": "Breakout jos"
        },
        {
            "prices": [100, 102, 98, 101, 99, 100, 101, 99, 100, 102],
            "current": 105,
            "desc": "Range mic"
        }
    ]
    
    results = []
    
    for test in test_cases:
        print(f"\n🔍 Test: {test['desc']}")
        
        # Ground truth (NumPy)
        z_true, mean, std = calculate_zscore_numpy(test['prices'], test['current'])
        
        # Prompt pentru model
        prompt = f"""Calculează Z-Score folosind formula: Z = (X - μ) / σ
Date prețuri: {test['prices']}
Preț curent (X): {test['current']}
Media (μ): {mean:.2f}
Deviație standard (σ): {std:.2f}

Returnază DOAR valoarea numerică a Z-Score."""
        
        inputs = tokenizer([prompt], return_tensors="pt").to("cuda")
        outputs = model.generate(**inputs, max_new_tokens=20, temperature=0.01)
        response = tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        # Extrage numărul din răspuns
        import re
        numbers = re.findall(r"-?\d+\.?\d*", response.split("assistant")[-1])
        
        if numbers:
            z_pred = float(numbers[0])
            error = abs(z_true - z_pred)
            error_percent = (error / abs(z_true)) * 100 if z_true != 0 else error * 100
            
            results.append({
                "test": test['desc'],
                "z_true": z_true,
                "z_pred": z_pred,
                "error": error,
                "error_percent": error_percent
            })
            
            print(f"  NumPy: {z_true:.6f}")
            print(f"  Model: {z_pred:.6f}")
            print(f"  Eroare: {error:.6f} ({error_percent:.2f}%)")
        else:
            print("  ⚠️ Modelul nu a returnat un număr valid")
    
    # Raport final
    print("\n" + "="*50)
    print("📊 RAPORT FINAL ACURATEȚE")
    print("="*50)
    
    if results:
        avg_error = np.mean([r['error_percent'] for r in results])
        print(f"Eroare medie: {avg_error:.2f}%")
        
        if avg_error < 5:
            print("✅ EXCELENT: Modelul păstrează matematica la 1.5 biți!")
        elif avg_error < 20:
            print("⚠️ ACCEPTABIL: Necesită IQ2_XS (2.0 biți) pentru matematică")
        else:
            print("❌ INSUFICIENT: Crește biții la 2.5 sau adaugă date de antrenare")
    
    return results

if __name__ == "__main__":
    results = test_model_accuracy()
    
    # Salvare rezultate
    with open("test_results.json", "w") as f:
        json.dump(results, f, indent=2)
    print("\n📁 Rezultate salvate în test_results.json")

# Script Ensemble - Rulează toate 4 modelele simultan
from unsloth import FastLanguageModel
import torch

class GridDSLEnsemble:
    def __init__(self):
        self.models = {}
        self.tokenizers = {}
        
        # Încărcăm cele 4 versiuni
        versions = {
            'ultra_fast': 'grid_dsl_1.5bit',   # 1.5 biți
            'balanced': 'grid_dsl_2.0bit',     # 2.0 biți
            'math_specialist': 'grid_dsl_2.5bit', # 2.5 biți
            'master': 'grid_dsl_4bit'          # 4.0 biți
        }
        
        for name, path in versions.items():
            print(f"📦 Încărc {name}...")
            model, tokenizer = FastLanguageModel.from_pretrained(
                path,
                load_in_4bit=True,
                device_map="auto"
            )
            FastLanguageModel.for_inference(model)
            self.models[name] = model
            self.tokenizers[name] = tokenizer
    
    def predict_ensemble(self, prompt, task_type="general"):
        """Predict cu toate modelele și alege cel mai bun"""
        
        results = {}
        
        for name, model in self.models.items():
            inputs = self.tokenizers[name]([prompt], return_tensors="pt").to("cuda")
            outputs = model.generate(**inputs, max_new_tokens=200)
            response = self.tokenizers[name].decode(outputs[0], skip_special_tokens=True)
            results[name] = response
        
        # Logică de unificare
        if task_type == "math":
            # Prioritizează modelul matematic
            return results['math_specialist']
        elif task_type == "code":
            # Voting între modele
            return max(results.values(), key=len)
        else:
            # Balans
            return results['balanced']
    
    def get_vram_usage(self):
        """Calculează VRAM total folosit"""
        total = 0
        for name, model in self.models.items():
            mem = sum(p.numel() * p.element_size() for p in model.parameters()) / 1024 / 1024
            print(f"{name}: {mem:.1f} MB")
            total += mem
        print(f"Total VRAM: {total:.1f} MB / 8192 MB")

# Exemplu utilizare
ensemble = GridDSLEnsemble()
ensemble.get_vram_usage()

# Test crypto
response = ensemble.predict_ensemble(
    "Calculează Z-Score pentru [62000, 62500, 61800] cu preț curent 63000",
    task_type="math"
)
print(response)
