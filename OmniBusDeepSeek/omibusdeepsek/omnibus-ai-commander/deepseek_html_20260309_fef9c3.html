Instalare Pachete KVM
Ubuntu/Debian
Instalare completă 
# Actualizează sistemul
sudo apt update
sudo apt upgrade -y

# Instalează KVM și pachetele necesare
sudo apt install -y qemu-kvm qemu-utils libvirt-daemon-system \
    libvirt-clients bridge-utils virt-manager ovmf \
    cpu-checker virt-viewer libguestfs-tools

# Pornește serviciile
sudo systemctl enable --now libvirtd
sudo systemctl enable --now virtlogd

# Adaugă utilizatorul în grupuri
sudo usermod -aG libvirt $USER
sudo usermod -aG kvm $USER

# Verifică instalarea
virsh list --all
systemctl status libvirtd

Configurare Rețea
Bridge sau NAT
Configurare bridge 
# Instalează bridge-utils
sudo apt install bridge-utils

# Crează bridge (ex: br0)
sudo nano /etc/netplan/01-netcfg.yaml

# Exemplu configurație:
network:
  version: 2
  ethernets:
    enp3s0:
      dhcp4: no
  bridges:
    br0:
      interfaces: [enp3s0]
      dhcp4: yes
      parameters:
        stp: false

# Aplică configurația
sudo netplan apply

Script Automat Complet

                            
Pentru Ubuntu 20.04/22.04

                        

                    


                    

                        

                            setup-gpu-passthrough.sh
                            
                        

                        
#!/bin/bash
# Script complet pentru GPU Passthrough
# Rulează cu: sudo bash setup-gpu-passthrough.sh

set -e

# Culori pentru output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

echo -e "${GREEN}🚀 Pornim configurarea GPU Passthrough...${NC}"

# Verificare root
if [[ $EUID -ne 0 ]]; then
   echo -e "${RED}Acest script trebuie rulat ca root (sudo)${NC}" 
   exit 1
fi

# Detectare CPU
if grep -q "vmx" /proc/cpuinfo; then
    CPU_TYPE="intel"
    IOMMU_OPTION="intel_iommu=on"
    echo -e "${GREEN}✅ CPU Intel detectat${NC}"
elif grep -q "svm" /proc/cpuinfo; then
    CPU_TYPE="amd"
    IOMMU_OPTION="amd_iommu=on"
    echo -e "${GREEN}✅ CPU AMD detectat${NC}"
else
    echo -e "${RED}❌ CPU nu suportă virtualizare hardware${NC}"
    exit 1
fi

# Solicită ID-urile GPU
echo -e "${YELLOW}Introduceți ID-urile GPU-ului (ex: 10de:2482,10de:228b):${NC}"
read GPU_IDS

# Backup GRUB
cp /etc/default/grub /etc/default/grub.backup

# Configurare GRUB
echo -e "${GREEN}🔧 Configurează GRUB...${NC}"
sed -i "s/GRUB_CMDLINE_LINUX_DEFAULT=\"/GRUB_CMDLINE_LINUX_DEFAULT=\"$IOMMU_OPTION iommu=pt /" /etc/default/grub
update-grub

# Blacklist drivere NVIDIA
echo -e "${GREEN}🔧 Crează blacklist...${NC}"
cat > /etc/modprobe.d/blacklist-nvidia.conf < /etc/modprobe.d/vfio.conf <> /etc/modules <

Troubleshooting

                        
Probleme comune și soluții

                    

                


                

                    

                        
!

                        

                            
GPU-ul nu apare în VM

                            
Verifică dacă ID-urile sunt corecte și dacă IOMMU este activat în BIOS

                        

                    


                    

                        
!

                        

                            
Eroare "No IOMMU detected"

                            
Activează în BIOS: Intel VT-d / AMD IOMMU și adaugă kernel parameters

                        

                    


                    

                        
!

                        

                            
Code 43 în Windows VM

                            
Adaugă hypervisor args în XML-ul VM-ului: <hyperv><relaxed state='on'/></hyperv>

Creare Mașină Virtuală

                            
Cu suport GPU Passthrough

                        

                    


                    

                        

                            virt-install command
                            
                        

                        
# Creare VM cu GPU Passthrough
sudo virt-install \
    --name grid-dsl-vm \
    --ram 16384 \
    --vcpus 8 \
    --disk path=/var/lib/libvirt/images/grid-dsl-vm.qcow2,size=100,format=qcow2 \
    --cdrom /path/to/ubuntu.iso \
    --os-variant ubuntu22.04 \
    --network network=default \
    --graphics spice \
    --video none \
    --host-device 03:00.0 \
    --host-device 03:00.1 \
    --boot uefi

                    


                    

                        ℹ️
                        Înlocuiește '03:00.0' și '03:00.1' cu adresele PCI ale GPU-ului tău

Configurație XML

                            
Pentru editare manuală

                        

                    


                    

                        

                            VM XML (virsh edit)
                            
                        

                        
<domain type='kvm'>
  <name>grid-dsl-vm</name>
  <memory unit='GiB'>16</memory>
  <vcpu>8</vcpu>
  <os>
    <type arch='x86_64' machine='pc-q35-6.2'>hvm</type>
    <loader readonly='yes' type='pflash'>/usr/share/OVMF/OVMF_CODE.fd</loader>
    <nvram>/var/lib/libvirt/qemu/nvram/grid-dsl-vm_VARS.fd</nvram>
  </os>
  <features>
    <acpi/>
    <apic/>
    <hyperv>
      <relaxed state='on'/>
      <vapic state='on'/>
      <spinlocks state='on' retries='8191'/>
    </hyperv>
  </features>
  <devices>
    <hostdev mode='subsystem' type='pci' managed='yes'>
      <source>
        <address domain='0x0000' bus='0x03' slot='0x00' function='0x0'/>
      </source>
    </hostdev>
    <hostdev mode='subsystem' type='pci' managed='yes'>
      <source>
        <address domain='0x0000' bus='0x03' slot='0x00' function='0x1'/>
      </source>
    </hostdev>
  </devices>
</domain>

Looking Glass

                            
Pentru afișare de înaltă performanță

                        

                    


                    

                        

                            Instalare Looking Glass
                            
                        

                        
# Pe gazdă
git clone https://github.com/gnif/LookingGlass.git
cd LookingGlass
mkdir build && cd build
cmake ../
make
sudo make install

# În VM (Windows)
# Descarcă și instalează driverul IVSHM de la:
# https://fedorapeople.org/groups/virt/virtio-win/direct-downloads/upstream-virtio/

# Adaugă în XML-ul VM-ului:
<shmem name='looking-glass'>
  <model type='ivshmem-plain'/>
  <size unit='M'>32</size>
</shmem>
