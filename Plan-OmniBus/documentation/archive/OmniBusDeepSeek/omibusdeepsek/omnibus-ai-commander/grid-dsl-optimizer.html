Z-Score Calculator 
# Strategie arbitraj Z-Score
def calculate_zscore(prices, current):
    import numpy as np
    prices = np.array(prices)
    mean = np.mean(prices)
    std = np.std(prices)
    zscore = (current - mean) / std
    
    if abs(zscore) > 2.5:
        return "SIGNAL: Overbought/Oversold"
    return f"Z-Score: {zscore:.2f}"
