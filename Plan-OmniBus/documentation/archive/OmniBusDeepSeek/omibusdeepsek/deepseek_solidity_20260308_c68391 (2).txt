contract OmniBusFlashLoan {
    IOmniBus public omnibus;
    
    function executeFlashLoan(
        address token,
        uint amount,
        bytes calldata params
    ) external {
        // 1. Request flash loan prin OmniBus
        omnibus.requestFlashLoan(token, amount);
        
        // 2. Execute arbitrage
        uint profit = performArbitrage(params);
        
        // 3. Repay loan + profit
        omnibus.repayFlashLoan(token, amount + profit);
    }
}
